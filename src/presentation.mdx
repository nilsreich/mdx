{`def fibonacci(n):
    """Generiere Fibonacci-Folge bis n Terme."""
    a, b = 0, 1
    result = []
    
    for _ in range(n):
        result.append(a)
        a, b = b, a + b
    
    return result

# Erste 10 Fibonacci-Zahlen

print(f"Fibonacci: {fib}")`}
